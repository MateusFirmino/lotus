lt_begin
	lt_create numNota;
	lt_create media;
	lt_create somaNota;
	lt_create nota;
	lt_create aux;

	lt_show ("insira a quantidade de notas do aluno:");
	lt_read (numNota);
	aux = numNota;

	lt_while (aux>0)
		lt_show ("insira uma nota:");
		lt_read (nota);

		somaNota = somaNota + nota;
		aux = aux - 1;
	lt_endwhile

	media = somaNota / numNota;

	lt_if (media < 6 )
		lt_show ("reprovado");

	lt_else
		lt_show	("aprovado");

	lt_endif
lt_exit
